import turtle

wn = turtle.Screen()
t = turtle.Turtle()

def square(x, y, size, color):
    t.penup()
    t.goto(x, y)
    t.pendown()
    t.color(color)
    t.begin_fill()
    for i in range(4):
        t.forward(size)
        t.left(90)
    t.end_fill()

def rotate():
    t.clear()
    square(-50, -50, 100, "white")
    square(-30, -30, 60, "black")
    t.left(10)
    wn.ontimer(rotate, 50)

rotate()
wn.mainloop()
